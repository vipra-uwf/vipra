214 [[nodiscard]] inline static auto is_coord_loc_compare(CAttributeValue value1, CAttributeValue value2)
215 -> bool {
216 return (value1.type == Type::COORD && value2.type == Type::LOCATION) ||
217 (value1.type == Type::LOCATION && value2.type == Type::COORD);
218 }

229 [[nodiscard]] inline static auto coord_loc_compare(CAttributeValue value1, CAttributeValue value2,
230 Simpack pack) -> bool {
231 if (value1.type == Type::COORD && value2.type == Type::LOCATION) {
232 return pack.get_context().locations[value2.as<VIPRA::idx>()].contains(value1.as<VIPRA::f3d>());
233 }
234
235 if (value2.type == Type::COORD && value1.type == Type::LOCATION) {
236 return pack.get_context().locations[value1.as<VIPRA::idx>()].contains(value2.as<VIPRA::f3d>());
237 }
238
239 return false;
240 }

250 [[nodiscard]] inline static auto is_equal(CAttributeValue value1, CAttributeValue value2, Simpack pack)
251 -> bool {
252 if (is_coord_loc_compare(value1, value2)) {
253 return coord_loc_compare(value1, value2, pack);
254 }
255
256 if (value1.type != value2.type) return false;
257
258 switch (value1.type) {
259 case Type::INVALID:
260 return false;
261 case Type::NUMBER:
262 return value1.as<float>() == value2.as<float>();
263 case Type::COORD:
264 return value1.as<VIPRA::f3d>() == value2.as<VIPRA::f3d>();
265 case Type::STATE:
266 return value1.as<stateUID>() == value2.as<stateUID>();
267 case Type::STATUS:
268 return value1.as<EventStatus>() == value2.as<EventStatus>();
269 case Type::LOCATION:
270 return value1.as<VIPRA::idx>() == value2.as<VIPRA::idx>();
271 break;
272 }
273
274 return false;
275 }

285 [[nodiscard]] inline static auto is_not_equal(CAttributeValue value1, CAttributeValue value2, Simpack pack)
286 -> bool {
287 return !is_equal(value1, value2, pack);

500 static inline void set_goal(Target target, Simpack pack, CAttributeValue value) {
501 auto& goals = pack.get_goals();
502 auto& context = pack.get_context();
503 const auto& pedset = pack.get_pedset();
504
505 if (value.type == Type::COORD) {
506 goals.changeEndGoal(target.targetIdx, pedset.getPedCoords(target.targetIdx), value.as<VIPRA::f3d>());
507 } else if (value.type == Type::LOCATION) {
508 // TODO: this doesn't take into account two pedestrains going to the same location
509 goals.changeEndGoal(target.targetIdx, pedset.getPedCoords(target.targetIdx),
510 context.locations[value.as<VIPRA::idx>()].random_point(context.engine));
511 }
512
513 value.type_check(Type::COORD);
514 }

524 static inline void set_velocity(Target target, Simpack pack, VIPRA::State& state, CAttributeValue value) {
525 value.type_check(Type::COORD);
526
527 state.velocities[target.targetIdx] = value.as<VIPRA::f3d>();
528
529 VIPRA::f3d originalPos = pack.get_pedset().getPedCoords(target.targetIdx);
530 state.coords[target.targetIdx] = originalPos + (state.velocities[target.targetIdx] * pack.dT);
531 }

545 static inline void scale_velocity(Target target, Simpack pack, VIPRA::State& state, CAttributeValue value) {
546 value.type_check(Type::NUMBER);
547
548 const NumericValue& scale = *static_cast<const NumericValue*>(value.value);
549 // TODO (rolland): get magnitude, scale that then apply to the unit vector
550 float scaleVal = scale.value(target.targetIdx);
551 auto tempvel = state.velocities[target.targetIdx];
552 tempvel *= scaleVal;
553 state.coords[target.targetIdx] = pack.get_pedset().getPedCoords(target.targetIdx) + (tempvel * pack.dT);
554 }
