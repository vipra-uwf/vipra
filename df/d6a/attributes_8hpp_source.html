75 inline static cAttributeValue getPedValue(Target target, Attribute attr, Simpack pack) {

102 inline static void setPedValue(Target target, Attribute attr, Simpack pack, VIPRA::State& state,

104 switch (attr) {
105 case Attribute::POSITION:
106 state.coords.at(target.targetIdx) = *static_cast<const VIPRA::pcoord*>(value);

108 case Attribute::VELOCITY:

117 return;

120 }

136 static inline void updateVelocity(Target target, Simpack pack, VIPRA::State& state, cAttributeValue value) {
137 state.velocities.at(target.targetIdx) = *static_cast<const VIPRA::veloc*>(value);
138 VIPRA::f3d originalPos = pack.pedSet.getPedCoords(target.targetIdx);
