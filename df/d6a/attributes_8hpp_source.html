166 [[nodiscard]] inline static auto equal(CAttributeValue value1, CAttributeValue value2) -> bool {
167 if (value1.type != value2.type) return false;
168
169 switch (value1.type) {
170 case Type::INVALID:
171 return false;
172 case Type::NUMBER:
173 return *static_cast<const float*>(value1.value) == *static_cast<const float*>(value2.value);
174 case Type::COORD:
175 return *static_cast<const VIPRA::f3d*>(value1.value) == *static_cast<const VIPRA::f3d*>(value2.value);
176 case Type::STATE:
177 return *static_cast<const BHVR::stateUID*>(value1.value) ==
178 *static_cast<const BHVR::stateUID*>(value2.value);
179 case Type::STATUS:
180 return *static_cast<const BHVR::EventStatus*>(value1.value) ==
181 *static_cast<const BHVR::EventStatus*>(value2.value);
182 }

397 static inline void set_velocity(Target target, Simpack pack, VIPRA::State& state, CAttributeValue value) {
398 value.type_check(Type::COORD);

400 state.velocities[target.targetIdx] = *static_cast<const VIPRA::veloc*>(value.value);

418 static inline void scale_velocity(Target target, Simpack pack, VIPRA::State& state, CAttributeValue value) {
419 value.type_check(Type::NUMBER);
