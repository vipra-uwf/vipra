180 inline static CAttributeValue getPedValue(Target target, Attribute attr, Simpack pack) {
181 switch (attr) {
182 case Attribute::POSITION:
183 return {Type::COORD, &pack.pedSet.getPedCoords(target.targetIdx)};
184 case Attribute::VELOCITY:
185 return {Type::COORD, &pack.pedSet.getPedVelocity(target.targetIdx)};
186 case Attribute::GOAL:
187 return {Type::COORD, &pack.goals.getEndGoal(target.targetIdx)};
188 case Attribute::STATE:

328 static inline void setVelocity(Target target, Simpack pack, VIPRA::State& state,
329 CAttributeValue value) {
330 value.typeCheck(Type::COORD);
331
332 state.velocities.at(target.targetIdx) =
