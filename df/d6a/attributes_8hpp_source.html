190 [[nodiscard]] inline static auto is_coord_loc_compare(CAttributeValue value1, CAttributeValue value2)
191 -> bool {
192 return (value1.type == Type::COORD && value2.type == Type::LOCATION) ||
193 (value1.type == Type::LOCATION && value2.type == Type::COORD);

205 [[nodiscard]] inline static auto coord_loc_compare(CAttributeValue value1, CAttributeValue value2,
206 Simpack pack) -> bool {
207 if (value1.type == Type::COORD && value2.type == Type::LOCATION) {
208 return pack.get_context().locations[value2.as<VIPRA::idx>()].contains(value1.as<VIPRA::f3d>());
209 }
210
211 if (value2.type == Type::COORD && value1.type == Type::LOCATION) {
212 return pack.get_context().locations[value1.as<VIPRA::idx>()].contains(value2.as<VIPRA::f3d>());
213 }
214
215 return false;
216 }
217
226 [[nodiscard]] inline static auto is_equal(CAttributeValue value1, CAttributeValue value2, Simpack pack)
227 -> bool {
228 if (is_coord_loc_compare(value1, value2)) {
229 return coord_loc_compare(value1, value2, pack);
230 }
231
232 if (value1.type != value2.type) return false;
233
234 switch (value1.type) {
235 case Type::INVALID:
236 return false;
237 case Type::NUMBER:
238 return value1.as<float>() == value2.as<float>();
239 case Type::COORD:
240 return value1.as<VIPRA::f3d>() == value2.as<VIPRA::f3d>();
241 case Type::STATE:
242 return value1.as<stateUID>() == value2.as<stateUID>();
243 case Type::STATUS:
244 return value1.as<EventStatus>() == value2.as<EventStatus>();
245 case Type::LOCATION:
246 return value1.as<VIPRA::idx>() == value2.as<VIPRA::idx>();
247 break;
248 }
249
250 return false;
251 }

476 static inline void set_goal(Target target, Simpack pack, CAttributeValue value) {
477 auto& goals = pack.get_goals();
478 auto& context = pack.get_context();
479 const auto& pedset = pack.get_pedset();
480
481 if (value.type == Type::COORD) {
482 goals.changeEndGoal(target.targetIdx, pedset.getPedCoords(target.targetIdx), value.as<VIPRA::f3d>());
483 } else if (value.type == Type::LOCATION) {
484 // TODO: this doesn't take into account two pedestrains going to the same location
485 goals.changeEndGoal(target.targetIdx, pedset.getPedCoords(target.targetIdx),
486 context.locations[value.as<VIPRA::idx>()].random_point(context.engine));
487 }
488
489 value.type_check(Type::COORD);
490 }

500 static inline void set_velocity(Target target, Simpack pack, VIPRA::State& state, CAttributeValue value) {
501 value.type_check(Type::COORD);
502
503 state.velocities[target.targetIdx] = value.as<VIPRA::f3d>();
504
505 VIPRA::f3d originalPos = pack.get_pedset().getPedCoords(target.targetIdx);
506 state.coords[target.targetIdx] = originalPos + (state.velocities[target.targetIdx] * pack.dT);
507 }

521 static inline void scale_velocity(Target target, Simpack pack, VIPRA::State& state, CAttributeValue value) {
522 value.type_check(Type::NUMBER);
523
524 const NumericValue& scale = *static_cast<const NumericValue*>(value.value);
525 // TODO (rolland): get magnitude, scale that then apply to the unit vector
526 float scaleVal = scale.value(target.targetIdx);
527 auto tempvel = state.velocities[target.targetIdx];
528 tempvel *= scaleVal;
529 state.coords[target.targetIdx] = pack.get_pedset().getPedCoords(target.targetIdx) + (tempvel * pack.dT);
530 }
