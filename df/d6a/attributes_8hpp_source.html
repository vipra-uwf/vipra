167 [[nodiscard]] inline static bool equal(CAttributeValue value1, CAttributeValue value2) {
168 if (value1.type != value2.type) return false;
169
170 switch (value1.type) {
171 case Type::INVALID:
172 return false;
173 case Type::NUMBER:
174 return *static_cast<const float*>(value1.value) ==
175 *static_cast<const float*>(value2.value);
176 case Type::COORD:
177 return *static_cast<const VIPRA::f3d*>(value1.value) ==
178 *static_cast<const VIPRA::f3d*>(value2.value);
179 case Type::STATE:
180 return *static_cast<const BHVR::stateUID*>(value1.value) ==
181 *static_cast<const BHVR::stateUID*>(value2.value);
182 case Type::STATUS:
183 return *static_cast<const BHVR::EventStatus*>(value1.value) ==
184 *static_cast<const BHVR::EventStatus*>(value2.value);
185 }
186 }
