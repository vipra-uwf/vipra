135 template <typename T>
136 [[nodiscard]] static auto findEventComponent(BehaviorParser::EventContext* ctx)
137 -> std::optional<T> {
138 for (const auto& attr : ctx->event_attribute()) {
139 // Name
140 if constexpr (std::is_same_v<T, std::string>)
141 if (attr->event_name()) return attr->event_name()->ID()->toString();
142
143 // Start Condition
144 if constexpr (std::is_same_v<T, evStart>)
145 if (attr->event_start()) return attr->event_start();
146
147 // End Condition
148 if constexpr (std::is_same_v<T, evEnd>)
149 if (attr->event_end()) return attr->event_end();
150 }
151
152 return std::nullopt;
153 }
