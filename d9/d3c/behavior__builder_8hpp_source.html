175 template <typename T>
176 [[nodiscard]] static auto findEventComponent(BehaviorParser::EventContext* ctx)
177 -> std::optional<T> {
178 for (const auto& attr : ctx->event_attribute()) {
179 // Name
180 if constexpr (std::is_same_v<T, std::string>)
181 if (attr->event_name()) return attr->event_name()->ID()->toString();
182
183 // Start Condition
184 if constexpr (std::is_same_v<T, evStart>)
185 if (attr->event_start()) return attr->event_start();
186
187 // End Condition
188 if constexpr (std::is_same_v<T, evEnd>)
189 if (attr->event_end()) return attr->event_end();
190 }
191
192 return std::nullopt;
193 }
